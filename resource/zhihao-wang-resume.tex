% SPACING TWEAKS
%   \resumeSubHeadingListStart: itemsep = gap between jobs/schools/projects
%   \resumeItemListStart:        itemsep = gap between bullet points (negative = tighter)
%                                topsep  = gap between job title row and first bullet
%   \resumeSubheading:           \vspace{-Xpt} at end = gap after the title/subtitle row

\newcommand{\resumeItem}[1]{
  \item\small{#1}
}
% ---- JOB / EDUCATION ENTRY: {Company}{Location}{Title}{Date} ----
\newcommand{\resumeSubheading}[4]{
  \vspace{-1pt}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & \small #2 \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-5pt}
}
% ---- CONTINUED ROLE (same company, no repeated name/location) ----
\newcommand{\resumeSubheadingContinued}[2]{
  \vspace{-8pt}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textit{\small#1} & \textit{\small #2} \\
    \end{tabular*}\vspace{-5pt}
}
\newcommand{\resumeProjectHeading}[2]{
  \item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \small#1 & \small #2 \\
    \end{tabular*}\vspace{-5pt}
}
% ---- Outer list (wraps jobs, schools, projects) ----
